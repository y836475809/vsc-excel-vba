Function GetFileList(Byval targetPath)
	Dim folder, file
	For Each folder In targetPath.SubFolders
		GetFileList(folder)
	Next
	For Each file In targetPath.files
        If InStr(file, ".bas") > 0 Then
            arr.Add file
        End If
        If InStr(file, ".cls") > 0 Then
            arr.Add file
        End If
	Next
End Function
